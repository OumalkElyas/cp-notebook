% --- Custom Command for Inserting Code ---
% Usage: \includecodesnippet{Language}{Title}{Description}{path/to/file.ext}
\newcommand{\includecodesnippet}[4]{
    % Space before the subsection
    \vspace{0.5em} 
    \subsection*{#2}
    \addcontentsline{toc}{subsection}{#2} 
    
    % The Description block (if you leave it empty {}, it just skips it)
    {\scriptsize #3 \par}
    
    % Space between title/description and the code
    \vspace{0.3em} 
    
    \inputminted{#1}{#4}
}
